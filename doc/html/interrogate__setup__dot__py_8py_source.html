00048 def _get_locations(pkgs, package_dir):
00049     """
00050     based on setuptools logic and the package_dir dict, builds a dict
00051     of location roots for each pkg in pkgs.
00052     See http://docs.python.org/distutils/setupscript.html
00053 
00054     :returns: a dict {pkgname: root} for each pkgname in pkgs (and each of their parents)
00055     """
00056     # package_dir contains a dict {package_name: relativepath}
00057     # Example {'': 'src', 'foo': 'lib', 'bar': 'lib2'}
00058     #
00059     # '' means where to look for any package unless a parent package
00060     # is listed so package bar.pot is expected at lib2/bar/pot,
00061     # whereas package sup.dee is expected at src/sup/dee
00062     #
00063     # if package_dir does not state anything about a package,
00064     # setuptool expects the package folder to be in the root of the
00065     # project
00066     locations = {}
00067     allprefix = package_dir.get('', '')
00068     for pkg in pkgs:
00069         parent_location = None
00070         splits = pkg.split('.')
00071         # we iterate over compound name from parent to child
00072         # so once we found parent, children just append to their parent
00073         for key_len in range(len(splits)):
00074             key = '.'.join(splits[:key_len + 1])
00075             if key not in locations:
00076                 if key in package_dir:
00077                     locations[key] = package_dir[key]
00078                 elif parent_location is not None:
00079                     locations[key] = parent_location
00080                 else:
00081                     locations[key] = allprefix
00082             parent_location = locations[key]
00083     return locations

00086 def generate_cmake_file(package_name, version, scripts, package_dir, pkgs):
00087     """
00088     Generates lines to add to a cmake file which will set variables
00089 
00090     :param version: str, format 'int.int.int'
00091     :param scripts: [list of str]: relative paths to scripts
00092     :param package_dir: {modulename: path}
00093     :pkgs: [list of str] python_packages declared in catkin package
00094     """
00095     prefix = '%s_SETUP_PY' % package_name
00096     result = []
00097     result.append(r'set(%s_VERSION "%s")' % (prefix, version))
00098     result.append(r'set(%s_SCRIPTS "%s")' % (prefix, ';'.join(scripts)))
00099 
00100     # Remove packages with '.' separators.
00101     #
00102     # setuptools allows specifying submodules in other folders than
00103     # their parent
00104     #
00105     # The symlink approach of catkin does not work with such submodules.
00106     # In the common case, this does not matter as the submodule is
00107     # within the containing module.  We verify this assumption, and if
00108     # it passes, we remove submodule packages.
00109     locations = _get_locations(pkgs, package_dir)
00110     for pkgname, location in locations.items():
00111         if not '.' in pkgname:
00112             continue
00113         splits = pkgname.split('.')
00114         # hack: ignore write-combining setup.py files for msg and srv files
00115         if splits[1] in ['msg', 'srv']:
00116             continue
00117         # check every child has the same root folder as its parent
00118         parent_name = '.'.join(splits[:1])
00119         if location != locations[parent_name]:
00120             raise RuntimeError(
00121                 "catkin_export_python does not support setup.py files that combine across multiple directories: %s in %s, %s in %s" % (pkgname, location, parent_name, locations[parent_name]))
00122 
00123     # If checks pass, remove all submodules
00124     pkgs = [p for p in pkgs if '.' not in p]
00125 
00126     resolved_pkgs = []
00127     for pkg in pkgs:
00128         resolved_pkgs += [os.path.join(locations[pkg], pkg)]
00129 
00130     result.append(r'set(%s_PACKAGES "%s")' % (prefix, ';'.join(pkgs)))
00131     result.append(r'set(%s_PACKAGE_DIRS "%s")' % (prefix, ';'.join(resolved_pkgs).replace("\\", "/")))
00132     return result
